template <typename ValueType>
constexpr auto optica::Bind(ValueType& value) constexpr noexcept

Sets BindProperty for option.

Template parameters
ValueType Type of holding bind
Parameters
value Value binded

Note

You cant create constexpr Option if your value isn't constexpr available. In this case create non-constexpr option or don't use BindProperty

static double speed = 5.0;

constexpr auto option = optica::Option<"speed", double> |
optica::Bind(speed);

// In this case it's okey because taking ref or pointer to static variable
// is possible in constexpr context

double height{};

auto option_height = optica::Option<"height", double> | optica::Bind(height);

// In case of non static variable you should choose non-constexpr version

template <typename... Ts, typename... Us>
constexpr auto optica::operator|(OptionBuilder<Ts...> left, OptionBuilder<Us...> right) constexpr noexcept

Pipe operator for accumulating properties.

Template parameters
Ts Properties inside lhf OptionBuilder
Us Properties inside rhf OptionBuilder
Parameters
left OptionBuilder<Ts...> Left operand must contain name and value
right OptionBuilder<Us...> Right operand
Returns OptionBuilder<Ts...,Us...> New option with properties from left and right operands

Remark

Left operand requires ValidOrderExpression and whole expression requires ValidPropertyExpression for strict ordering in construction option

// Option1 is valid because Opt func sets Name and ValueType
auto Option1 = optica::Opt<"Day", int>() | optica::Required();
// Option2 is invalid because starts from Required
auto Option2 = optica::Required() | optica::Opt<"Day", int>;
